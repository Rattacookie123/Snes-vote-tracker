-- ========================================= -- 1) TABLES -- ========================================= create table if not exists public.games ( id bigint generated by default as identity primary key, name text not null unique, status text not null default 'todo' check (status in ('todo', 'finished', 'excluded')), vote_count integer not null default 0, created_at timestamptz not null default now() ); create table if not exists public.votes ( id bigint generated by default as identity primary key, game_id bigint not null references public.games(id) on delete cascade, user_id uuid not null, created_at timestamptz not null default now(), unique (game_id, user_id) ); create table if not exists public.admins ( user_id uuid primary key, created_at timestamptz not null default now() ); alter table public.games enable row level security; alter table public.votes enable row level security; alter table public.admins enable row level security; -- ========================================= -- 2) BASIC READ POLICIES -- ========================================= drop policy if exists "games_read_all" on public.games; create policy "games_read_all" on public.games for select to public using (true); drop policy if exists "admins_read_own" on public.admins; create policy "admins_read_own" on public.admins for select to authenticated using (auth.uid() = user_id); -- We do NOT allow direct vote inserts/updates/deletes from the client. -- That all goes through SQL functions below. -- ========================================= -- 3) HELPER: IS ADMIN? -- ========================================= create or replace function public.is_admin() returns boolean language sql security definer set search_path = public as $$ select exists ( select 1 from public.admins where user_id = auth.uid() ); $$; grant execute on function public.is_admin() to anon, authenticated; -- ========================================= -- 4) GET GAMES + WHETHER CURRENT USER VOTED -- ========================================= create or replace function public.get_games_with_votes() returns table ( name text, status text, vote_count integer, user_voted boolean ) language sql security definer set search_path = public as $$ select g.name, g.status, g.vote_count, exists ( select 1 from public.votes v where v.game_id = g.id and v.user_id = auth.uid() ) as user_voted from public.games g order by g.name asc; $$; grant execute on function public.get_games_with_votes() to anon, authenticated; -- ========================================= -- 5) TOGGLE VOTE (SAFE) -- one click = add vote -- second click = remove vote -- ========================================= create or replace function public.toggle_vote(p_game_name text) returns json language plpgsql security definer set search_path = public as $$ declare current_user_id uuid; target_game_id bigint; already_voted boolean; new_vote_count integer; begin current_user_id := auth.uid(); if current_user_id is null then raise exception 'Not authenticated'; end if; select id into target_game_id from public.games where name = p_game_name; if target_game_id is null then raise exception 'Game not found'; end if; select exists( select 1 from public.votes where game_id = target_game_id and user_id = current_user_id ) into already_voted; if already_voted then delete from public.votes where game_id = target_game_id and user_id = current_user_id; update public.games set vote_count = greatest(vote_count - 1, 0) where id = target_game_id; else insert into public.votes (game_id, user_id) values (target_game_id, current_user_id); update public.games set vote_count = vote_count + 1 where id = target_game_id; end if; select vote_count into new_vote_count from public.games where id = target_game_id; return json_build_object( 'success', true, 'user_voted', not already_voted, 'vote_count', new_vote_count ); end; $$; grant execute on function public.toggle_vote(text) to anon, authenticated; -- ========================================= -- 6) ADMIN: SET STATUS -- ========================================= create or replace function public.set_game_status(p_game_name text, p_status text) returns void language plpgsql security definer set search_path = public as $$ begin if not public.is_admin() then raise exception 'Not authorized'; end if; if p_status not in ('todo', 'finished', 'excluded') then raise exception 'Invalid status'; end if; update public.games set status = p_status where name = p_game_name; end; $$; grant execute on function public.set_game_status(text, text) to authenticated; -- ========================================= -- 7) ADMIN: RESET ALL VOTES -- ========================================= create or replace function public.reset_all_votes() returns void language plpgsql security definer set search_path = public as $$ begin if not public.is_admin() then raise exception 'Not authorized'; end if; delete from public.votes; update public.games set vote_count = 0; end; $$; grant execute on function public.reset_all_votes() to authenticated; -- ========================================= -- 8) SEED GAMES -- ========================================= insert into public.games (name, status, vote_count) values ('Chrono Trigger', 'todo', 0), ('Super Metroid', 'todo', 0), ('Terranigma', 'todo', 0), ('Secret of Mana', 'todo', 0), ('ActRaiser', 'todo', 0) on conflict (name) do nothing;
